```ts /** * @file cart.ts * @description Component to display the user's shopping cart. * Shows the items in the cart, quantities, prices, and total cost. * Allows users to update quantities or remove items. * Provides a button to proceed to checkout. */ // Import necessary modules (none needed for this example, but kept for potential expansion) /** * Function to render the shopping cart. * @returns {string} - HTML string representing the shopping cart. */ function renderCart() { // Retrieve cart data from local storage const cart = JSON.parse(localStorage.getItem('cart') || '[]'); // Calculate total price let totalPrice = 0; for (const item of cart) { totalPrice += item.price * item.quantity; } // Function to handle quantity changes function updateQuantity(productId: string, newQuantity: number) { const updatedCart = cart.map(item => { if (item.id === productId) { return { ...item, quantity: newQuantity }; } return item; }); // Remove item if quantity is zero const filteredCart = updatedCart.filter(item => item.quantity > 0); localStorage.setItem('cart', JSON.stringify(filteredCart)); // Re-render the cart document.getElementById('cart-container').innerHTML = renderCart(); } // Function to remove item from cart function removeItem(productId: string) { const updatedCart = cart.filter(item => item.id !== productId); localStorage.setItem('cart', JSON.stringify(updatedCart)); // Re-render the cart document.getElementById('cart-container').innerHTML = renderCart(); } // Build the HTML for the cart items let cartItemsHTML = cart.map(item => `

${item.name}

$${item.price}

`).join(''); // If the cart is empty, display a message if (cart.length === 0) { cartItemsHTML = '

Your cart is empty.

'; } // Build the complete HTML for the cart const cartHTML = `

Shopping Cart

${cartItemsHTML}

Total:

$${totalPrice.toFixed(2)}

`; return cartHTML; } // Function to initialize the cart component function initCart() { const cartContainer = document.getElementById('cart-container'); if (cartContainer) { cartContainer.innerHTML = renderCart(); } else { console.error('Cart container element not found.'); } } // Call the initCart function when the page loads document.addEventListener('DOMContentLoaded', initCart); // Export the initCart function (if needed for other modules) export { initCart }; ```
